skl2onnx                    # scikit-learn → ONNX

# For Hugging Face transformer models
optimum[onnxruntime]        # CPU
optimum[onnxruntime-gpu]    # GPU   

# For quantization
onnxruntime                 # quantization tools are bundled in onnxruntime
